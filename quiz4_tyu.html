# import data
healthcare.spending = head(read.csv("healthcare-spending.csv", header = TRUE, sep = ",", skip = 2), -9)
healthcare.spending = subset(healthcare.spending, Location != "District of Columbia" & Location != "United States")
healthcare.spending$Location = as.character(healthcare.spending$Location)
healthcare.spending[healthcare.spending$Location == "District of Columbia",] = NULL
region = read.csv("federalRegions.txt", header = FALSE, sep = "", skip = 1)
region$V2 = as.character(region$V2)
region$V3 = as.character(region$V3)
region$Location = trimws(paste(region$V2, region$V3), which = "right")
region$V2 = NULL
region$V3 = NULL
colnames(region) = c("Region", "Location")
region$Region = as.factor(region$Region)

# merge data
joined = left_join(region, healthcare.spending, by = "Location")
